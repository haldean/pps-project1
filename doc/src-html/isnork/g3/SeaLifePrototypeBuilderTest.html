001    package isnork.g3;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.junit.Assert.assertFalse;
005    
006    import org.junit.Test;
007    
008    import isnork.sim.SeaLifePrototype;
009    
010    
011    public class SeaLifePrototypeBuilderTest {
012    
013            @Test
014            public void testProperties(){
015                    boolean dangerous = true;
016                    int happiness = 10;
017                    int max = 20;
018                    int min = 3;
019                    SeaLifePrototype builderCharmander = new SeaLifePrototypeBuilder("charmander")
020                            .dangerous(dangerous)
021                            .happiness(happiness)
022                            .maxCount(max)
023                            .minCount(min)
024                            .create();
025                    
026                    SeaLifePrototype nonBuilderCharmander = new SeaLifePrototype();
027                    nonBuilderCharmander.setDangerous(dangerous);
028                    nonBuilderCharmander.setHappiness(happiness);
029                    nonBuilderCharmander.setMaxCount(max);
030                    nonBuilderCharmander.setMinCount(min);
031                    nonBuilderCharmander.setName("charmander");
032                    
033                    assertEquals(nonBuilderCharmander.isDangerous(), builderCharmander.isDangerous());
034                    assertEquals(nonBuilderCharmander.getHappiness(), builderCharmander.getHappiness());
035                    assertEquals(nonBuilderCharmander.getMaxCount(), builderCharmander.getMaxCount());
036                    assertEquals(nonBuilderCharmander.getMinCount(), builderCharmander.getMinCount());
037                    assertEquals(nonBuilderCharmander.getName(), builderCharmander.getName());
038            }
039    
040    
041            @Test
042            public void testDoubleCreation(){
043                    SeaLifePrototypeBuilder builder = new SeaLifePrototypeBuilder("charmander");
044                    SeaLifePrototype firstCharmander = builder.create();
045                    SeaLifePrototype secondCharmander = builder.create();
046                    
047                    assertFalse(firstCharmander == secondCharmander);               
048            }
049            
050            @Test
051            public void testDoubleCreationModification(){
052                    SeaLifePrototypeBuilder builder = new SeaLifePrototypeBuilder("charmander");
053                    SeaLifePrototype charmander = builder.name("charmander").create();
054                    SeaLifePrototype squirtle = builder.name("squirtle").create();
055                    
056                    assertEquals("charmander", charmander.getName());               
057                    assertEquals("squirtle", squirtle.getName());           
058            }
059    }
