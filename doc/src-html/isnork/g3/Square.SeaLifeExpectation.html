001    package isnork.g3;
002    
003    import java.util.Set;
004    
005    import isnork.sim.Player;
006    import isnork.sim.SeaLife;
007    
008    public interface Square {
009      public interface SeaLifeExpectation {
010        SeaLife getSeaLife();
011        double getCertainty();
012      }
013    
014      /**
015       * Add the presence of a creature to the square, along with the certainty
016       * that it is on the square.
017       */
018            void addCreature(SeaLife creature, double certainty);
019    
020      /**
021       * Get the creatures thought to be on this square, along with the certainty of
022       * its presence.
023       */
024            Set<SeaLifeExpectation> getCreatures();
025    
026      /**
027       * Add the presence of a diver to this square.
028       */
029            void addDiver(Player diver);
030    
031      /**
032       * Get the divers currently on this square.
033       */
034            Set<Player> getDivers();
035    
036      /**
037       * Set the expected danger value for this square.
038       */
039            void setExpectedDanger(double danger);
040    
041      /**
042       * Get the expected danger value for this square.
043       */
044            double getExpectedDanger();
045    
046      /**
047       * Set the expected happiness value for this square. Note: this includes
048       * subtracting the danger associated with this square.
049       */
050            void setExpectedHappiness(double happiness);
051    
052      /**
053       * Get the expected happiness value for this square. Note: this includes
054       * subtracting the danger associated with this square.
055       */
056            double getExpectedHappiness();
057    
058      /**
059       * Reset the square. This should be called at the start of each tick, and
060       * should remove all happiness on the square that is not due to the existence
061       * of stationary creatures. It should also remove all divers and
062       * nonstationary creatures from the corresponding sets.
063       */
064      void clear();
065    }
