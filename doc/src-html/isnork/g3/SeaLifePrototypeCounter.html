001    package isnork.g3;
002    
003    import isnork.sim.SeaLifePrototype;
004    
005    /**
006     * 
007     * @author Moses Nakamura
008     * Prefer composition to inheritance blah blah blah
009     */
010    
011    public class SeaLifePrototypeCounter {
012            private int count;
013            private SeaLifePrototype delegate;
014            private int captured;
015            
016            public SeaLifePrototypeCounter(SeaLifePrototype prototype){
017                    this.delegate = prototype;
018                    this.count = 0;
019                    this.captured = 0;
020            }
021            
022            /**
023             * Should be called whenever a creature is seen
024             */
025            public void sawCreature(){
026                    count++;
027            }
028            
029            /**
030             * Fetches the number of times a given species has been seen
031             */
032            public int getSeenCount(){
033                    return count;
034            }
035            
036            /**
037             * Fetches the actual prototype
038             * @return
039             */
040            public SeaLifePrototype get(){
041                    return delegate;
042            }
043    
044            /**
045             * Whether the species is dangerous
046             * @return
047             */
048            public boolean isDangerous() {
049                    return delegate.isDangerous();
050            }
051    
052            /**
053             * Mincount on the species
054             * @return
055             */
056            public int getMinCount() {
057                    return delegate.getMinCount();
058            }
059    
060            /**
061             * Maxcount on the species
062             * @return
063             */
064            public int getMaxCount() {
065                    return delegate.getMaxCount();
066            }
067    
068            /**
069             * The happiness of the species
070             * @return
071             */
072            public int getHappiness() {
073                    return delegate.getHappiness();
074            }
075    
076            /**
077             * The number of creatures you have gotten happiness from
078             */
079            public void captured() {
080                    this.captured++;
081            }
082            
083            /**
084             * The number of creatures you have gotten happiness from
085             * @return 
086             */
087            public int getCaptured() {
088                    return captured;
089            }
090    }
