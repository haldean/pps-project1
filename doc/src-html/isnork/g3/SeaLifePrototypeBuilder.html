001    package isnork.g3;
002    
003    import isnork.sim.SeaLifePrototype;
004    
005    /**
006     * Builder for SeaLifePrototype, to instantiate hit create
007     * Default settings are happiness 1, mincount 1, maxcount 1, 
008     * dangerous false.
009     * Name field MUST be set.  If you try to create while you have
010     * a trivial name, you will get an IllegalStateException in your face
011     * @author Moses
012     *
013     */
014    
015    public class SeaLifePrototypeBuilder {
016    
017            private SeaLifePrototype prototype;
018            private String name;
019            private int happiness;
020            private int minCount;
021            private boolean dangerous;
022            private int maxCount;
023    
024            public SeaLifePrototypeBuilder(){
025                    happiness = 1;
026                    minCount = 1;
027                    maxCount = 1;
028                    dangerous = false;
029                    name = "";
030            }
031            
032            public SeaLifePrototypeBuilder(String name){
033                    happiness = 1;
034                    minCount = 1;
035                    maxCount = 1;
036                    dangerous = false;
037                    this.name = name;
038            }
039            
040            public SeaLifePrototypeBuilder dangerous(boolean dangerous){
041                    this.dangerous = dangerous;
042                    return this;
043            }
044            
045            public SeaLifePrototypeBuilder maxCount(int maxCount){
046                    this.maxCount = maxCount;
047                    return this;
048            }
049            
050            public SeaLifePrototypeBuilder minCount(int minCount){
051                    this.minCount = minCount;
052                    return this;
053            }
054    
055            public SeaLifePrototypeBuilder happiness(int happiness){
056                    this.happiness = happiness;
057                    return this;
058            }
059    
060            public SeaLifePrototypeBuilder name(String name){
061                    this.name = name;
062                    return this;
063            }
064            
065            public SeaLifePrototype create(){
066                    if (name == ""){
067                            throw new IllegalStateException("You must have a non-trivial name.");
068                    }
069                    prototype = new SeaLifePrototype();
070                    prototype.setDangerous(dangerous);
071                    prototype.setHappiness(happiness);
072                    prototype.setMaxCount(maxCount);
073                    prototype.setMinCount(minCount);
074                    prototype.setName(name);
075                    return prototype;
076            }
077    }
