001    package isnork.g3;
002    
003    import isnork.sim.SeaLife;
004    import isnork.sim.Player;
005    import java.util.Set;
006    import com.google.common.collect.*;
007    
008    /**
009     * Records what creatures, divers, the danger, and happiness of a square
010     */
011    public class WaterProofSquare implements Square {
012            private Set<Square.SeaLifeExpectation> creatures = Sets.newHashSet();
013            private Set<Player> divers = Sets.newHashSet();
014            private double danger = 0.0;
015            private double happiness = 0.0;
016    
017            public void addCreature(SeaLife creature, double certainty) {
018                    creatures.add(new ExpectedCreature(creature, certainty));
019            }
020    
021            public Set<Square.SeaLifeExpectation> getCreatures() {
022                    return creatures;
023            }
024    
025            public void addDiver(Player diver) {
026                    divers.add(diver);
027            }
028    
029            public Set<Player> getDivers() {
030                    return divers;
031            }
032    
033            public void setExpectedDanger(double danger) {
034                    this.danger = danger;
035            }
036    
037            public double getExpectedDanger() {
038                    return danger;
039            }
040    
041            public void setExpectedHappiness(double happiness) {
042                    this.happiness = happiness;
043            }
044    
045            public double getExpectedHappiness() {
046                    return happiness;
047            }
048    
049      public void clear() {
050        danger = 0.;
051        happiness = 0.;
052        divers.clear();
053    
054        for (SeaLifeExpectation creature : creatures) {
055          SeaLife seaLife = creature.getSeaLife();
056          if (true /* TODO(haldean): NO. */) {
057            creatures.remove(creature);
058          } else {
059            happiness += seaLife.getHappiness();
060            if (seaLife.isDangerous()) {
061              danger -= 2 * happiness;
062            }
063          }
064        }
065      }
066    
067      public class ExpectedCreature implements Square.SeaLifeExpectation {
068        private SeaLife creature;
069        private double certainty;
070    
071        public ExpectedCreature(SeaLife creature, double certainty) {
072          this.creature = creature;
073          this.certainty = certainty;
074        }
075    
076        public SeaLife getSeaLife() {
077          return creature;
078        }
079    
080        public double getCertainty() {
081          return certainty;
082        }
083      }
084    }
