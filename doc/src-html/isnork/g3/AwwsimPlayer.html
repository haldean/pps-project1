053             * Process input, determine message to be sent

055            @Override
056            public String tick(Point2D myPosition, Set<Observation> whatYouSee,
057                            Set<iSnorkMessage> incomingMessages,
058                            Set<Observation> playerLocations) {
059        ticks++;
060        currentLocation = myPosition;
061                    carto.update(myPosition,whatYouSee,playerLocations,incomingMessages); 
062                    carto.getNextDirection();
063                    return carto.getMessage();
064            }

066            /**
067             * Move should be stored based on information in tick, then return it
068             */
069            @Override
070            public Direction getMove() {
071        if (ticks < MAX_TICKS_PER_ROUND - 100) {
072          Direction[] dirs = Direction.values();
073          return dirs[random.nextInt(dirs.length)];
074        } else {
075          // Move towards boat
076          int deltaX = 0;
077          int deltaY = 0;
078          if (currentLocation.getX() < 0)
079            deltaX = 1;
080          else if (currentLocation.getX() > 0)
081            deltaX = -1;
082    
083          if (currentLocation.getY() < 0)
084            deltaY = 1;
085          else if (currentLocation.getY() > 0)
086            deltaY = -1;
087    
088          if (deltaX > 0 && deltaY == 0) {
089            return Direction.E;
090          } else if (deltaX > 0 && deltaY > 0) {
091            return Direction.NE;
092          } else if (deltaX > 0 && deltaY < 0) {
093            return Direction.SE;
094          } else if (deltaX == 0 && deltaY == 0) {
095            return Direction.STAYPUT;
096          } else if (deltaX == 0 && deltaY < 0) {
097            return Direction.N;
098          } else if (deltaX == 0 && deltaY > 0) {
099            return Direction.S;
100          } else if (deltaX < 0 && deltaY == 0) {
101            return Direction.W;
102          } else if (deltaX < 0 && deltaY > 0) {
103            return Direction.NW;
104          } else if (deltaX < 0 && deltaY < 0) {
105            return Direction.NE;
106          } else {
107            throw new RuntimeException("I HAVE NO IDEA");
108          }
109        }
110            }
