001    package isnork.g3;
002    import isnork.sim.GameObject.Direction;
003    import isnork.sim.Observation;
004    import isnork.sim.iSnorkMessage;
005    
006    import java.awt.geom.Point2D;
007    import java.util.Set;
008    
009    public class WaterProofCartogram implements Cartogram {
010      private final int sideLength;
011            private final int viewRadius;
012            private final int numDivers;
013            private final Square[][] mapStructure;
014    
015            public WaterProofCartogram(int mapWidth, int viewRadius, int numDivers) {
016                    this.sideLength = mapWidth;
017                    this.viewRadius = viewRadius;
018                    this.numDivers = numDivers;
019                    this.mapStructure = new WaterProofSquare[sideLength][sideLength];
020            }
021    
022            @Override
023            public String update(Point2D myPosition, Set<Observation> whatYouSee,
024                            Set<Observation> playerLocations,
025                            Set<iSnorkMessage> incomingMessages) {
026                    for (Observation location : playerLocations) {
027                            location.getLocation();
028                            location.getId();
029                            location.getName();
030                    }
031                    
032                    for (iSnorkMessage message : incomingMessages) {
033                            message.getLocation();
034                            message.getMsg();
035                            message.getSender();
036                    }
037                    
038                    for (Observation observation : whatYouSee) {
039                            observation.getDirection();
040                            observation.getLocation();
041                            observation.getId();
042                            observation.happiness();
043                            observation.isDangerous();
044                            observation.happinessD();
045                            observation.getName();
046                    }
047    
048                    return null;
049            }
050    
051            @Override
052            public String getMessage() {
053        return "";
054            }
055    
056            @Override
057            public Direction getNextDirection() {
058        return null;
059            }
060    }
