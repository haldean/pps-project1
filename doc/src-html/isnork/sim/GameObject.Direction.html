001    package isnork.sim;
002    
003    import java.awt.geom.Point2D;
004    import java.util.ArrayList;
005    import java.util.HashSet;
006    
007    public abstract class GameObject {
008            protected double x;
009            protected double y;
010            public enum Direction{
011                    E(0,false,1,0),NE(45,true,1,-1),N(90,false,0,-1),NW(135,true,-1,-1),W(180,false,-1,0),SW(225,true,-1,1),S(270,false,0,1),SE(315,true,1,1),STAYPUT(0,false,0,0);
012                    int deg;
013                    boolean diag;
014                    public int dx;
015                    public int dy;
016                    public int getDx() {
017                            return dx;
018                    }
019                    public int getDy() {
020                            return dy;
021                    }
022                    Direction(int degrees, boolean diag,int x,int y)
023                    {
024                            deg = degrees;
025                            this.diag = diag;
026                            this.dx=x;
027                            this.dy=y;
028                    }
029                    public int getDegrees() {
030                            return deg;
031                    }
032                    public boolean isDiag()
033                    {
034                            return diag;
035                    }
036                    public static ArrayList<Direction> allBut(Direction n)
037                    {
038                            ArrayList<Direction> r = new ArrayList<GameObject.Direction>();
039                            for(Direction d : Direction.values())
040                            {
041                                    if(n == null || d.deg != n.deg )
042                                            r.add(d);
043                            }
044                            return r;
045                    }
046            }
047            public Point2D getLocation()
048            {
049                    return new Point2D.Double(x,y);
050            }
051            public double getX() {
052                    return x;
053            }
054            public double getY() {
055                    return y;
056            }
057            public void setX(double x) {
058                    this.x = x;
059            }
060            public void setY(double y) {
061                    this.y = y;
062            }
063    }
