001    package isnork.sim;
002    
003    import java.awt.Color;
004    import java.awt.Point;
005    import java.awt.geom.Point2D;
006    import java.util.ArrayList;
007    import java.util.HashSet;
008    import java.util.Iterator;
009    import java.util.Set;
010    
011    import org.w3c.dom.Document;
012    import org.w3c.dom.Element;
013    
014    public class Cell
015    {
016            private Boolean Observed;
017            private Double ObservedDistance;
018            private Boolean Stepped;
019            private int stepcount;
020            private ArrayList<CellListener> listeners;
021            
022    
023            /*Cell bounds*/
024            private Point Location;
025            private Point2D.Double Center;
026            private Double Dimensions;
027            
028            
029            /*Cell Properties*/
030            private Boolean blocked;
031            private Boolean startable;
032            
033            HashSet<SeaLife> sealife = new HashSet<SeaLife>();
034            HashSet<Integer> players = new HashSet<Integer>();
035    
036            public Set<SeaLife> getSeaLife()
037            {
038                    Set<SeaLife> ret = new HashSet<SeaLife>();
039                    for(SeaLife s : sealife)
040                    {
041                            ret.add((SeaLife) s.clone());
042                    }
043                    return sealife;
044            }
045            public Set<Integer> getPlayers()
046            {
047                    return players;
048            }
049            public Cell(int x, int y, Double dimensions)
050            {
051                    CellBounds(new Point(x, y), dimensions);
052                    Stepped = false;
053                    stepcount = 0;
054                    Observed = false;
055                    ObservedDistance = Double.MAX_VALUE;
056                    
057                    listeners = new ArrayList<CellListener>();
058            }
059    
060            public Cell(Point location, Double dimensions)
061            {
062                    CellBounds(location, dimensions);
063                    Stepped = false;
064                    stepcount = 0;
065                    Observed = false;
066                    ObservedDistance = Double.MAX_VALUE;
067    
068                    listeners = new ArrayList<CellListener>();
069            }
070    
071            public void CellBounds(Point location, Double dimensions)
072            {
073                    this.setLocation(location.x, location.y);
074                    Dimensions = dimensions;
075                    blocked = false;
076                    startable = false;
077                    
078                    Center = new Point2D.Double(((double)Location.x + 0.5d)*Dimensions, ((double)Location.y + 0.5d)*Dimensions);
079            }
080            
081    
082            /**
083             * @return
084             */
085            
086            public Boolean isSteppedOn()
087            {
088                    // Do something here
089                    return Stepped;
090            }
091            
092            public Boolean getObserved()
093            {
094                    return new Boolean(Observed);
095            }
096    
097            /**
098             * @return
099             */
100            public Double getObservedDistance()
101            {
102                    return new Double(ObservedDistance);
103            }
104            
105    
106            /**
107             * @param c
108             * @return
109             */
110            public Double distanceToCell(Cell c)
111            {
112                    return Center.distance(c.Center);
113            }
114            
115            /**
116             * @param c
117             * @return
118             */
119            public Cell copy(Cell c)
120            {
121                    return new Cell(c.Location, c.Dimensions);
122            }
123    
124            /**
125             * @return
126             */
127            public Point getLocation()
128            {
129                    return new Point(Location);
130            }
131    
132            /**
133             * @return
134             */
135            public Point2D.Double getCenter()
136            {
137                    return new Point2D.Double(Center.x,Center.y);
138            }
139    
140            /**
141             * @return
142             */
143            public Double getDimensions()
144            {
145                    return new Double(Dimensions);
146            }
147    
148    
149            
150            public void setLocation(int x, int y)
151            {
152                    Location = new Point(x, y);
153            }
154    
155            
156            public Boolean isBlocked()
157            {
158                    return new Boolean(blocked);
159            }
160            
161            public Boolean isStartable()
162            {
163                    return new Boolean(startable);
164            }
165    
166            public void setStartable(Boolean startable)
167            {
168                    this.startable = startable;
169                    notifyListeners();
170            }
171            
172            public static interface CellListener{
173                    public void CellUpdated(Cell source);
174            }
175    
176            public void addCellListener(CellListener l){
177                    listeners.add(l);
178            }
179            public void removeCellListener(CellListener l){
180                    listeners.remove(l);
181            }
182            private void notifyListeners(){
183                    Iterator<CellListener> it = listeners.iterator();
184                    while(it.hasNext())
185                            it.next().CellUpdated(this);
186            }
187            
188            @Override
189            public String toString()
190            {
191                    return "(" + Location.x + "," + Location.y + ")";
192            }
193            
194    }
