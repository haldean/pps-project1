169            public boolean step() {
170                    if(round > 480)
171                    {
172                            Point2D origin = new Point2D.Double(0, 0);
173                            dsq = 0;
174                            for (Player pl : players) {
175                                    if(!pl.location.equals(origin))
176                                    {
177                                            pl.happiness-=config.getPenalty();
178                                            score-=config.getPenalty();
179                                            dsq++;
180                                    }
181                            }
182                            notifyListeners(GameUpdateType.GAMEOVER);
183                            return false;
184                    }
185                    HashSet<SeaLife> dequeue = new HashSet<SeaLife>();
186                    for (QueuedSeaLife qs : queuedSealife.values()) {
187                            qs.ttl--;
188                            if (qs.ttl == 0) {
189                                    SeaLife s = qs.sealife;
190                                    Direction d = qs.d;
191                                    int dist = 1;
192                                    // Do the move.
193                                    Point2D p = new Point2D.Double(s.getLocation().getX() + d.dx, s
194                                                    .getLocation().getY() + d.dy);
195                                    s.setLocation(p);
196                                    s.setDirection(d);
197                                    dequeue.add(qs.sealife);
198                            }
199                    }
200                    for (SeaLife s : dequeue)
201                            queuedSealife.remove(s);
202    
203                    HashSet<Player> dequeuep = new HashSet<Player>();
204                    for (QueuedPlayer qs : queuedPlayers.values()) {
205                            qs.ttl--;
206                            if (qs.ttl == 0) {
207                                    Player pl = qs.player;
208                                    Direction d = qs.d;
209                                    int dist = 1;
210                                    // Do the move.
211                                    Point2D p = new Point2D.Double(pl.location.getX() + d.dx,
212                                                    pl.location.getY() + d.dy);
213                                    pl.location = p;
214                                    dequeuep.add(qs.player);
215                            }
216                    }
217                    for (Player s : dequeuep)
218                            queuedPlayers.remove(s);
219    
220                    for (SeaLife s : config.creatures) {
221                            if (s.speed > 0 && !queuedSealife.containsKey(s)) {
222                                    // Do the move thing.
223                                    Direction d = s.getNewDirection();
224                                    Point2D p = new Point2D.Double(s.getLocation().getX() + d.dx, s
225                                                    .getLocation().getY() + d.dy);
226                                    while (Math.abs(p.getX()) > GameConfig.d
227                                                    || Math.abs(p.getY()) > GameConfig.d) {
228                                            d = s.getNewDirection();
229                                            p = new Point2D.Double(s.getLocation().getX() + d.dx, s
230                                                            .getLocation().getY() + d.dy);
231                                    }
232                                    QueuedSeaLife q = new QueuedSeaLife();
233                                    q.d = d;
234                                    q.sealife = s;
235                                    if (d.diag)
236                                            q.ttl = 3;
237                                    else
238                                            q.ttl = 2;
239                                    queuedSealife.put(s, q);
240                            }
241                    }
242                    HashSet<iSnorkMessage> newmsg = new HashSet<iSnorkMessage>();
243                    for (Player pl : players) {
244                            HashSet<Observation> observations = new HashSet<Observation>();
245                            for (SeaLife s : config.creatures) {
246                                    if(s.getLocation().distance(pl.location) <= config.r)
247                                    {
248                                            if(!beenSeenById.containsKey(pl))
249                                                    beenSeenById.put(pl, new HashSet<SeaLife>());
250                                            if(!beenSeenByName.containsKey(pl))
251                                                    beenSeenByName.put(pl, new HashMap<String, Integer>());
252                                            double happy = 0;
253                                            
254                                            if(!pl.location.equals(new Point2D.Double(0, 0))) { // if on the boat, doesn't affect
255                                                if(s.dangerous && s.getLocation().distance(pl.location) <= 1.5)
256                                                    happy = happy - s.happiness*2;
257                                                else if(beenSeenById.get(pl).contains(s))
258                                                    happy = 0;
259                                                else if(!beenSeenByName.get(pl).containsKey(s.getName()))
260                                                {
261                                                    happy = s.happiness;
262                                                    beenSeenByName.get(pl).put(s.getName(), 1);
263                                                }
264                                                else if(beenSeenByName.get(pl).get(s.getName()) == 1)
265                                                {
266                                                    if(!pl.location.equals(new Point2D.Double(0, 0))) {
267                                                        happy = s.happiness/2;
268                                                        beenSeenByName.get(pl).put(s.getName(), 2);                                         
269                                                    }
270                                                }
271                                                else if(beenSeenByName.get(pl).get(s.getName()) == 2 && !pl.location.equals(new Point2D.Double(0, 0)))
272                                                {
273                                                    happy = s.happiness/4;
274                                                    beenSeenByName.get(pl).put(s.getName(), 3);
275                                                }
276    
277                                                beenSeenById.get(pl).add(s);
278                                                pl.happiness += happy;
279                                                score += happy;
280                                            }
281                                            
282                                            
283                                            Observation o = new Observation();
284                                            o.id=s.getId();
285                                            o.name=s.getName();
286                                            o.dir=s.getDirection();
287                                            o.location=(Point2D) s.getLocation().clone();
288                                            o.happy=happy;
289                                            o.danger=s.dangerous;
290                                            observations.add(o);
291    
292                                    }
293                            }
294                            HashSet<Observation> locations = new HashSet<Observation>();
295                            for (Player pp : players) {
296                                    if(pp != pl && pp.location.distance(pl.location) <= config.r)
297                                    {
298                                            Observation o = new Observation();
299                                            o.id=pp.getId();
300                                            o.name=pp.getName();
301                                            o.location=(Point2D) pp.location.clone();
302                                            o.happy=0;
303                                            o.danger=false;
304                                            observations.add(o);
305                                    }
306                                    Observation o = new Observation();
307                                    o.id=pp.getId();
308                                    o.name=pp.getName();
309                                    o.location=(Point2D) pp.location.clone();
310                                    o.happy=0;
311                                    o.danger=false;
312                                    locations.add(o);
313                            }
314                            
315                            String msg = null;
316                            try
317                            {
318    
319                                    msg = pl.tick((Point2D) pl.location.clone(), observations, messages, locations);
320                            }
321                            catch(Exception e)
322                            {
323                                    exceptioned = true;
324                                    round = 481;
325                                    return true;
326                            }
327                            
328                            if (msg != null){
329                                    iSnorkMessage m = new iSnorkMessage(msg);
330                                    m.sender=pl.id;
331                                    m.location=pl.location;
332                                    newmsg.add(m);
333                            }
334                            
335                            if (!queuedPlayers.containsKey(pl)) {
336                                    Direction d = null;
337                                    try
338                                    {
339                                            d = pl.getMove();
340                                    }
341                                    catch(Exception e)
342                                    {
343                                            System.err.println("Timeout/exception");
344                                            exceptioned = true;
345                                            round = 481;
346                                            return true;
347                                    }
348                                    if (d != null) {
349                                            Point2D p = new Point2D.Double(pl.location.getX() + d.dx,
350                                                            pl.location.getY() + d.dy);
351                                            if (Math.abs(p.getX()) > GameConfig.d
352                                                            || Math.abs(p.getY()) > GameConfig.d) {
353                                                    GameEngine.println("SIM>>Player " + pl.id
354                                                                    + " returned an invalid move");
355                                            } else {
356                                                    QueuedPlayer q = new QueuedPlayer();
357                                                    q.player = pl;
358                                                    q.d = d;
359                                                    q.ttl = (d.diag ? 3 : 2);
360                                                    queuedPlayers.put(pl, q);
361                                            }
362                                    }
363                            }
364                    }
365                    messages = newmsg;
366                    notifyListeners(GameUpdateType.MOVEPROCESSED);
367                    round++;
368                    return true;
369            }

617                    }
