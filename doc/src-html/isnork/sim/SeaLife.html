001    package isnork.sim;
002    
003    import isnork.sim.GameObject.Direction;
004    
005    import java.awt.geom.Point2D;
006    import java.util.ArrayList;
007    
008    public class SeaLife extends SeaLifePrototype implements Cloneable {
009            private int id;
010            private Point2D location;
011            private Direction direction;
012            
013            public void setId(int id) {
014                    this.id = id;
015            }
016            public int getId() {
017                    return id;
018            }
019            public Point2D getLocation() {
020                    return location;
021            }
022            public void setLocation(Point2D location) {
023                    this.location = location;
024            }
025            public Direction getDirection() {
026                    return direction;
027            }
028            public void setDirection(Direction direction) {
029                    this.direction = direction;
030            }
031            public Direction getNewDirection() {
032                    int r = GameConfig.random.nextInt(100);
033                    if(r < 21 || direction == null)
034                    {
035                            ArrayList<Direction> directions = Direction.allBut(direction);
036                            return directions.get(GameConfig.random.nextInt(directions.size()));
037                    }
038                    else
039                    {
040                            //Keep same direction
041                            return direction;
042                    }
043            }
044            public SeaLife(SeaLifePrototype p)
045            {
046                    this.happiness=p.happiness;
047                    this.dangerous=p.dangerous;
048                    this.name=p.name;
049                    this.speed=p.speed;
050                    this.filename=p.filename;
051            }
052            public Object clone()
053            {
054                    SeaLife s = new SeaLife(this);
055                    s.id=this.id;
056                    s.location=this.location;
057                    s.direction= null;
058                    return s;
059            }
060            @Override
061            public String toString() {
062                    return name+" (id "+id+", happy score = "+happiness+(dangerous?", dangerous" : "")+")";
063            }
064    }
