077            public void paint(Graphics g) {
078                    if (board == null)
079                            return;
080                    int w = (int) Board.toScreenSpaceNoOffset(1);
081                    if (w != cacheWidth) {
082                            cacheWidth = w;
083                            icons = new HashMap<String, Image>();
084                    }
085                    int[][] counts = new int[GameConfig.d * 2 + 1][GameConfig.d * 2 + 1];
086                    for (int i = 0; i < counts.length; i++) {
087                            for (int j = 0; j < counts.length; j++)
088                                    counts[i][j] = 0;
089                    }
090                    Graphics2D g2D = (Graphics2D) g;
091                    g2D.setColor(ocrean_color);
092                    boardBox = new Rectangle2D.Double(
093                                    Board.toScreenSpace(GameConfig.d * -1),
094                                    Board.toScreenSpace(GameConfig.d * -1),
095                                    Board.toScreenSpace(GameConfig.d + 1),
096                                    Board.toScreenSpace(GameConfig.d + 1));
097                    g2D.fill(boardBox);
098                    g2D.setColor(Color.black);
099                    g2D.fillOval((int) Board.toScreenSpace(0),
100                                    (int) Board.toScreenSpace(0),
101                                    (int) Board.toScreenSpaceNoOffset(1),
102                                    (int) Board.toScreenSpaceNoOffset(1));
103                    
104                    if(onPlayer)
105                    {
106                            g2D.setColor(visible_color);
107                            double tx = Board.toScreenSpace(MouseCoords.getX() - engine.getConfig().getR());
108                            double ty = Board.toScreenSpace(MouseCoords.getY()- engine.getConfig().getR() );
109                            Ellipse2D visibleBox = new Ellipse2D.Double(tx,ty, Board.toScreenSpaceNoOffset(engine.getConfig().getR()*2 + 1) , Board.toScreenSpaceNoOffset(engine.getConfig().getR()*2 + 1) );
110                            g2D.fill(visibleBox);
111                    }
112                    g2D.setColor(Color.black);
113                    drawGrid(g2D);
114    
115                    if (engine != null && engine.getConfig() != null
116                                    && engine.getConfig().creatures != null) {
117                            for (SeaLife s : engine.getConfig().creatures) {
118                                    counts[(int) (s.getLocation().getX() + GameConfig.d)][(int) (s
119                                                    .getLocation().getY() + GameConfig.d)]++;
120                                    if (s.filename != null) {
121    
122                                            if (!icons.containsKey(s.filename)) {
123                                                    BufferedImage img = null;
124                                                    try {
125                                                            img = ImageIO.read(new File("boards/icons/" + s.filename));
126                                                            icons.put(s.filename, img.getScaledInstance(w, w,
127                                                                            BufferedImage.SCALE_FAST));
128                                                    } catch (IOException ioe) {
129                                                            GameEngine.println("Unable to read image icons/"
130                                                                            + s.filename);
131                                                            ioe.printStackTrace();
132                                                    }
133                                            }
134                                            g2D.drawImage(icons.get(s.filename),
135                                                            (int) Board.toScreenSpace(s.getLocation().getX()),
136                                                            (int) Board.toScreenSpace(s.getLocation().getY()),
137                                                            null);
138    
139                                    } else {
140                                            Ellipse2D d = new Ellipse2D.Double(Board.toScreenSpace(s
141                                                            .getLocation().getX()), Board.toScreenSpace(s
142                                                            .getLocation().getY()),
143                                                            Board.toScreenSpaceNoOffset(1),
144                                                            Board.toScreenSpaceNoOffset(1));
145                                            g2D.fill(d);
146                                    }
147                            }
148                            for (Player p : engine.players) {
149                                    counts[(int) (p.location.getX() + GameConfig.d)][(int) (p.location
150                                                    .getY() + GameConfig.d)]++;
151                                    BufferedImage img = null;
152                                    try {
153                                            img = ImageIO.read(new File("boards/icons/diver.png"));
154                                            icons.put("diver", img.getScaledInstance(w, w,
155                                                            BufferedImage.SCALE_FAST));
156                                    } catch (IOException ioe) {
157                                            ioe.printStackTrace();
158                                            GameEngine.println("Unable to read image icons/diver.png");
159                                    }
160                                    g2D.drawImage(icons.get("diver"),
161                                                    (int) Board.toScreenSpace(p.location.getX()),
162                                                    (int) Board.toScreenSpace(p.location.getY()), null);
163    
164                            }
165                            for (int i = 0; i < counts.length; i++) {
166                                    for (int j = 0; j < counts.length; j++)
167                                            if (counts[i][j] > 1) {
168                                                    g2D.drawString("" + counts[i][j],
169                                                                    (float) Board.toScreenSpaceNoOffset(i),
170                                                                    (float) Board.toScreenSpaceNoOffset(j) + w);
171                                            }
172                            }
173                    }
174            }

176            private void drawGrid(Graphics2D g2d) {
177                    g2d.setColor(Color.black);
178                    for (int i = -GameConfig.d; i <= GameConfig.d + 1; i++) {
179                            Line2D line = new Line2D.Double(0, Board.toScreenSpace(i),
180                                            Board.toScreenSpace(GameConfig.d + 1),
181                                            Board.toScreenSpace(i));
182                            g2d.draw(line);
183                            line = new Line2D.Double(Board.toScreenSpace(i), 0,
184                                            Board.toScreenSpace(i),
185                                            Board.toScreenSpace(GameConfig.d + 1));
186                            g2d.draw(line);
187                    }
188            }

212            private int floorAbs(double v) {
213                    int r = (int) Math.abs(v);
214                    if (v < 0)
215                            r = 0 - r - 1;
216                    return r;
217            }
218            boolean onPlayer = false;
219            Point mouseLoc = null;
220            public void mouseMoved(MouseEvent e) {
221                    MouseCoords = Board.fromScreenSpace(e.getPoint());
222                    MouseCoords = new Point2D.Double(floorAbs(MouseCoords.getX()),
223                                    floorAbs(MouseCoords.getY()));
224                    if (engine != null)
225                            engine.mouseChanged();
226                    String tip = "<html><b>Location:</b> (" + MouseCoords.getX() + ", " + MouseCoords.getY() + ")"+(MouseCoords.getX()==MouseCoords.getY() && MouseCoords.getX()==0 ? " <b>BOAT</B>" : "")+"<br><b>Sealife here:</b><br>";
227                    int n=0;
228                    if(engine != null && engine.getConfig() != null)
229                    {
230                            for(SeaLife s : engine.getConfig().creatures)
231                            {
232                                    if(s.getLocation().equals(MouseCoords))
233                                    {
234                                            tip+=s.toString()+"<br>";
235                                            n++;
236                                    }
237                            }
238                            if(n==0)
239                                    tip+="<i>No sealife here</i><br>";
240                            n =0;
241                            tip +="<b>Players here:</b><br>";
242                            onPlayer = false;
243                            for(Player p : engine.players)
244                            {
245                                    if(p.location.equals(MouseCoords))
246                                    {
247                                            onPlayer = true;
248                                            tip+=p.toString()+"<br>";
249                                            n++;
250                                    }
251                            }
252                            if(n==0)
253                                    tip+="<i>No players here</i><br>";
254                    }
255                    tip += "</html>";
256                    this.setToolTipText(tip);
257                    repaint();
258            }

266            public void recalculateDimensions() {
267                    int my_w = this.getWidth();
268                    int my_h = this.getHeight();
269                    int d = Math.min(my_w, my_h);
270                    d -= 10;
271                    if (d > 0)
272                            Board.pixels_per_pixel = ((double) GameConfig.d * 2 + 1)
273                                            * Board.pixels_per_meter / d;
274                    repaint();
275            }
