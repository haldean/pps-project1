076      @SuppressWarnings("unchecked")
077            public ConfigurationPanel(GameConfig config) {
078                    this.config = config;
079    
080                    this.setBorder(BorderFactory.createTitledBorder(
081                                    BorderFactory.createEtchedBorder(), "Configuration"));
082                    this.setPreferredSize(new Dimension(350, 1200));
083                    
084                    this.setLayout(new GridLayout(0, 2));
085    
086    
087    
088                    numAntsLabel = new JLabel("D (Dimension): ");
089                    numAntsLabel.setFont(config_font);
090                    dSpinner = new JSpinner(new SpinnerNumberModel(
091                                    GameConfig.d, 1, null, 1));
092                    dSpinner.setPreferredSize(new Dimension(100, 10));
093                    dSpinner.addChangeListener(this);
094    
095                    add(numAntsLabel);
096                    add(dSpinner);
097    
098    
099                    JLabel lbl = new JLabel("R: ");
100                    lbl.setFont(config_font);
101                    rSpinner = new JSpinner(new SpinnerNumberModel(
102                                    this.config.getR(), 1, null, 1));
103                    rSpinner.setPreferredSize(new Dimension(100, 10));
104                    rSpinner.addChangeListener(this);
105    
106                    add(lbl);
107                    add(rSpinner);
108    
109                    JLabel label = new JLabel("N (Number of Divers): ");
110                    label.setFont(config_font);
111                    numDiversSpinner = new JSpinner(new SpinnerNumberModel(this.config.getNumDivers(), 1, null,
112                                    1));
113                    numDiversSpinner.setPreferredSize(new Dimension(100, 10));
114                    numDiversSpinner.addChangeListener(this);
115                    add(label);
116                    add(numDiversSpinner);
117    
118                    randomSeed = new JSpinner(new SpinnerNumberModel(this.config.getRandomSeed(), new Long(1), null,
119                                    new Long(1)));
120                    randomSeed.addChangeListener(this);
121                    add(new JLabel("Random seed"));
122                    add(randomSeed);
123    
124                    playerLabel = new JLabel("Player:");
125                    add(playerLabel);
126                    // make player combo box
127                    playerBox = new JComboBox(config.getPlayerListModel());
128                    playerBox.addItemListener(this);
129                    playerBox.setRenderer(new ClassRenderer());
130                    add(playerBox);
131    
132                    config.setPlayerClass((Class<Player>) playerBox.getSelectedItem());
133    
134                    // board combo
135                    boardLabel = new JLabel("Board:");
136                    boardBox = new JComboBox(config.getBoardList());
137                    boardBox.addItemListener(this);
138                    add(boardLabel);
139                    add(boardBox);
140    
141                    speedSlider = new JSlider(0, 1000);
142                    speedSlider.setValue(0);
143                    add(new JLabel("Delay (0 - 1000ms):"));
144                    add(speedSlider);
145    
146                    mouseCoords = new JLabel("Mouse: N/A");
147                    score = new JLabel("Score: N/A");
148                    add(score);
149                    add(mouseCoords);
150    
151            }
152    
153            public void setMouseCoords(Point2D p) {
154                    if (p == null)
155                            mouseCoords.setText("Mouse: N/A");
156                    float v1 = (float) Math.round(p.getX() * 100) / 100;
157                    float v2 = (float) Math.round(p.getY() * 100) / 100;
158                    mouseCoords.setText(String.format("Mouse: %4.2f, %4.2f", v1, v2));
159            }
160    
161            protected JLabel mouseCoords;
162    
163            public void setEnabled(boolean enabled) {
164                    super.setEnabled(enabled);
165                    boardBox.setEnabled(enabled);
166                    numDiversSpinner.setEnabled(enabled);
167                    dSpinner.setEnabled(enabled);
168                    playerBox.setEnabled(enabled);
169            }
170    
171            public void stateChanged(ChangeEvent arg0) {
172                     if (arg0.getSource().equals(rSpinner))
173                     config.setR(((Integer) ((JSpinner)
174                     arg0.getSource()).getValue()).intValue());
175                     else if(arg0.getSource().equals(dSpinner))
176                     config.setD(((Integer) ((JSpinner)
177                     arg0.getSource()).getValue()).intValue());
178                     else if(arg0.getSource().equals(numDiversSpinner))
179                     config.setNumDivers(((Integer) ((JSpinner)
180                     arg0.getSource()).getValue()).intValue());
181                     else if(arg0.getSource().equals(randomSeed))
182                             config.setRandomSeed(((Long) ((JSpinner)
183                             arg0.getSource()).getValue()).longValue());
184                     else
185                     throw new RuntimeException("Unknown State Changed Event!!");
186            }
187    
188      @SuppressWarnings("unchecked")
189            public void itemStateChanged(ItemEvent arg0) {
190                    if (arg0.getSource().equals(playerBox)
191                                    && arg0.getStateChange() == ItemEvent.SELECTED) {
192                            // config.setActivePlayer((Class)arg0.getItem());
193                            selectedPlayer = (Class) arg0.getItem();
194                            config.setPlayerClass((Class<Player>) playerBox.getSelectedItem());
195    
196                    }
197                    if (arg0.getSource().equals(boardBox)
198                                    && arg0.getStateChange() == ItemEvent.SELECTED) {
199                            config.setSelectedBoard((File) arg0.getItem());
200                    }
201            }
202    
203            public void reloadBoards() {
204                    boardBox.removeAllItems();
205                    File[] files = config.getBoardList();
206                    for (int i = 0; i < files.length; i++)
207                            boardBox.addItem(files[i]);
208            }

211                    return speedSlider;
212            }

215            }
216    
217            public void updateScores(double s) {
218                    score.setText("Score: " + s);
219            }
