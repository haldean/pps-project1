385            private class XMLFilter extends FileFilter {
386    
387                    @Override
388                    public boolean accept(File f) {
389                            return f.getName().toLowerCase().endsWith(".xml");
390                    }
391    
392                    @Override
393                    public String getDescription() {
394                            return "XML Files";
395                    }
396    
397            }

399            public void actionPerformed(ActionEvent arg0) {
400                    // find our super parent frame -- needed for dialogs
401                    Component c = this;
402                    while (null != c.getParent())
403                            c = c.getParent();
404    
405                    if (arg0.getActionCommand().equalsIgnoreCase("SAVE")) {
406                            chooser.setCurrentDirectory(new File("boards"));
407                            int returnVal = chooser.showSaveDialog(this);
408                            if (returnVal == JFileChooser.APPROVE_OPTION) {
409                                    File file = chooser.getSelectedFile();
410                                    try {
411                                            XMLEncoder d = new XMLEncoder(new FileOutputStream(
412                                                            chooser.getSelectedFile()));
413                                            for(SeaLifePrototype t : board)
414                                            {
415                                                    d.writeObject(t);
416                                            }
417                                            d.close();
418                                            JOptionPane.showMessageDialog((Frame) c,
419                                                            "File saved successfully.", "Success",
420                                                            JOptionPane.INFORMATION_MESSAGE);
421                                    } catch (Exception e) {
422                                            JOptionPane.showMessageDialog((Frame) c, e, "Save Error",
423                                                            JOptionPane.ERROR_MESSAGE);
424                                    }
425                            }
426                    } else if (arg0.getActionCommand().equalsIgnoreCase("LOAD")) {
427                            chooser.setCurrentDirectory(new File("boards"));
428                            int returnVal = chooser.showOpenDialog(this);
429    
430                            if (returnVal == JFileChooser.APPROVE_OPTION) {
431    
432                                    loading = true;
433    
434                                    try {
435                                            XMLDecoder d = new XMLDecoder(new FileInputStream(
436                                                            chooser.getSelectedFile()));
437                                            board= new ArrayList<SeaLifePrototype>();
438                                            SeaLifePrototype n = (SeaLifePrototype) d.readObject();
439                                            while (n != null) {
440                                                    board.add(n);
441                                                    try {
442                                                            n = (SeaLifePrototype) d.readObject();
443                                                    } catch (ArrayIndexOutOfBoundsException e) {
444                                                            n = null;
445                                                    }
446                                            }
447                                    } catch (IOException e) {
448                                            System.err.println("Error reading configuration file:"
449                                                            + e.getMessage());
450                                            e.printStackTrace();
451                                            System.exit(-1);
452                                    }
453    
454                                    setBoard(board);
455                                    JOptionPane.showMessageDialog((Frame) c,
456                                                    "File loaded successfully.", "Success",
457                                                    JOptionPane.INFORMATION_MESSAGE);
458                                    loading = false;
459                                    // }catch(IOException e){
460                                    // // e.printStackTrace();
461                                    // JOptionPane.showMessageDialog( (Frame)c, e, "Load Error",
462                                    // JOptionPane.ERROR_MESSAGE);
463                                    // loading = false;
464                                    // }
465                            }
466                    } else if (arg0.getActionCommand().equalsIgnoreCase("RESET")) {
467                            board = new ArrayList<SeaLifePrototype>();
468                            boardPanel.setBoard(board);
469                            this.repaint();
470                            boardPanel.repaint();
471                    } else if (arg0.getActionCommand().equalsIgnoreCase("ADD")) {
472                            boardPanel.addNewSealifePrototype();
473                    }
474            }
