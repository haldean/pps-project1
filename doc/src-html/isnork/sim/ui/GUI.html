223            private class GameRunner extends Thread implements ChangeListener {
224                    private JSlider slider;
225                    private int delay;
226    
227                    public GameRunner(JSlider slider) {
228                            this.slider = slider;
229                    }
230    
231                    public void run() {
232                            delay = slider.getValue();
233                            slider.addChangeListener(this);
234                            while (fast && engine.step()) {
235                                    try {
236                                            Thread.sleep(delay);
237                                    } catch (InterruptedException e) {
238                                            // this should not happen!
239                                            e.printStackTrace();
240                                    }
241                            }
242                            slider.removeChangeListener(this);
243                    }
244    
245                    public void stateChanged(ChangeEvent arg0) {
246                            if (arg0.getSource().equals(slider)) {
247                                    delay = ((JSlider) arg0.getSource()).getValue();
248                            }
249                    }
250            }

252            private String toHumanTime(int t) {
253                    String r = "";
254                    int h = 0;
255                    int m = 0;
256                    h = t / 60;
257                    m = t % 60;
258                    if (h > 0)
259                            r = h + "h, ";
260                    r += m + "m";
261                    return r;
262            }

264            public void gameUpdated(GameUpdateType type) {
265                    // find our super parent frame -- needed for dialogs
266                    Component c = this;
267                    while (null != c.getParent())
268                            c = c.getParent();
269    
270                    switch (type) {
271                    case GAMEOVER:
272                            fast = false;
273                            boardPanel.repaint();
274                             configPanel.updateScores(engine.getScores());
275                            controlPanel.play.setEnabled(false);
276                            controlPanel.step.setEnabled(false);
277                            controlPanel.pause.setEnabled(false);
278                            controlPanel.stop.setEnabled(false);
279                            controlPanel.begin.setEnabled(true);
280                            configPanel.setEnabled(true);
281                            String s = "After 8 hours, total happiness: "
282                                            + engine.getScores() + ", avg: "+(engine.getScores()/engine.getConfig().getNumDivers()
283                                                            + " DSQ: " + engine.getDsq());
284                            JOptionPane.showMessageDialog((Frame) c, s, "Game Over",
285                                            JOptionPane.INFORMATION_MESSAGE);
286    
287                            break;
288                    case MOVEPROCESSED:
289                            controlPanel.roundText
290                                            .setText(toHumanTime(engine.getCurrentRound()));
291                            boardFrame.round.setText("Elapsed: "
292                                            + toHumanTime(engine.getCurrentRound()));
293                            boardFrame.bp.repaint();
294                            boardPanel.repaint();
295                             configPanel.updateScores(engine.getScores());
296                            break;
297                    case STARTING:
298                            controlPanel.roundText.setText("0");
299                            boardPanel.recalculateDimensions();
300                            break;
301                    case MOUSEMOVED:
302                            configPanel.setMouseCoords(BoardPanel.MouseCoords);
303          break;
304                    default:
305                            // nothing.
306                    }
307            }
