001    package isnork.sim.ui;
002    
003    import isnork.sim.Player;
004    
005    import java.awt.Color;
006    import java.awt.Component;
007    
008    import javax.swing.DefaultListCellRenderer;
009    import javax.swing.JList;
010    
011    
012    public class ClassRenderer extends DefaultListCellRenderer
013    {
014      private static final long serialVersionUID = 25123L;
015    
016      @SuppressWarnings("unchecked")
017            public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus)
018            {
019                    try
020                    {
021                            super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
022                            if (value instanceof Class)
023                            {
024                                    Class<Player> p = (Class<Player>) value;
025                                    Player pl = p.newInstance();
026                                    String strPlayer = pl.getName();
027                                    if (strPlayer != null)
028                                    {
029                                            setText(strPlayer);
030                                    }
031                            }
032                    } catch (Exception e)
033                    {
034    
035                    }
036                    return this;
037            }
038    
039    }
