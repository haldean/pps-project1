007    package isnork.sim.ui;
008    
009    import java.awt.Dimension;
010    import java.awt.event.ActionListener;
011    
012    import javax.swing.JButton;
013    import javax.swing.JLabel;
014    import javax.swing.JPanel;
015    import javax.swing.JTextField;
016    
017    public final class ControlPanel extends JPanel{
018            /**
019             * 
020             */
021            private static final long serialVersionUID = 1L;
022            protected JButton begin;
023            protected JButton step;
024            protected JButton play;
025            protected JButton pause; 
026            protected JButton stop;
027            protected JButton tournament;
028            protected JTextField roundText;
029            protected JButton boardPanel;
030            
031            protected ControlPanel(){
032                    begin = new JButton("Begin New Game");
033                    begin.setName("Begin");
034                    begin.setActionCommand("Begin");
035                    begin.setEnabled(true);
036                    
037                    step = new JButton("Step");
038                    step.setName("Step");
039                    step.setEnabled(false);
040                    step.setActionCommand("Step");
041                    
042                    play = new JButton("Play");
043                    play.setName("Play");
044                    play.setEnabled(false);
045                    play.setActionCommand("Play");
046                    
047                    pause = new JButton("Pause");
048                    pause.setName("Pause");
049                    pause.setEnabled(false);
050                    pause.setActionCommand("Pause");
051                    
052                    stop = new JButton("Resign");
053                    stop.setName("Stop");
054                    stop.setEnabled(false);
055                    stop.setActionCommand("Stop");
056                    
057                    add(begin);
058                    add(step);
059                    add(play);
060                    add(pause);
061                    add(stop);
062                    
063                    JLabel label = new JLabel("Clock: ");
064                    roundText = new JTextField();
065                    roundText.setEnabled(false);
066                    roundText.setEditable(false);
067                    roundText.setPreferredSize(new Dimension(100, 25));
068                    add(label);
069                    add(roundText);
070                    
071            }
072            protected void addListener(ActionListener a){
073                    begin.addActionListener(a);
074                    pause.addActionListener(a);
075                    play.addActionListener(a);
076                    step.addActionListener(a);
077                    stop.addActionListener(a);
078            }
079    }
