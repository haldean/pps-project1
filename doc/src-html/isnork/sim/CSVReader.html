001    package isnork.sim;
002    
003    
004    import java.io.BufferedReader;
005    import java.io.IOException;
006    import java.io.Reader;
007    import java.util.ArrayList;
008    import java.util.List;
009    
010    /**
011     * A very simple CSV reader released under a commercial-friendly license.
012     *
013     * @author Glen Smith
014     *
015     */
016    public class CSVReader {
017    
018        private BufferedReader br;
019    
020        private boolean hasNext = true;
021    
022        private char separator;
023    
024        private char quotechar;
025    
026        private int skipLines;
027    
028        private boolean linesSkiped;
029    
030        /** The default separator to use if none is supplied to the constructor. */
031        public static final char DEFAULT_SEPARATOR = ',';
032    
033        /**
034         * The default quote character to use if none is supplied to the
035         * constructor.
036         */
037        public static final char DEFAULT_QUOTE_CHARACTER = '"';
038    
039        /**
040         * The default line to start reading.
041         */
042        public static final int DEFAULT_SKIP_LINES = 0;
043    
044        /**
045         * Constructs CSVReader using a comma for the separator.
046         *
047         * @param reader
048         *            the reader to an underlying CSV source.
049         */
050        public CSVReader(Reader reader) {
051            this(reader, DEFAULT_SEPARATOR);
052        }
053    
054        /**
055         * Constructs CSVReader with supplied separator.
056         *
057         * @param reader
058         *            the reader to an underlying CSV source.
059         * @param separator
060         *            the delimiter to use for separating entries.
061         */
062        public CSVReader(Reader reader, char separator) {
063            this(reader, separator, DEFAULT_QUOTE_CHARACTER);
064        }
065    
066    
067    
068        /**
069         * Constructs CSVReader with supplied separator and quote char.
070         *
071         * @param reader
072         *            the reader to an underlying CSV source.
073         * @param separator
074         *            the delimiter to use for separating entries
075         * @param quotechar
076         *            the character to use for quoted elements
077         */
078        public CSVReader(Reader reader, char separator, char quotechar) {
079            this(reader, separator, quotechar, DEFAULT_SKIP_LINES);
080        }
081    
082        /**
083         * Constructs CSVReader with supplied separator and quote char.
084         *
085         * @param reader
086         *            the reader to an underlying CSV source.
087         * @param separator
088         *            the delimiter to use for separating entries
089         * @param quotechar
090         *            the character to use for quoted elements
091         * @param line
092         *            the line number to skip for start reading
093         */
094        public CSVReader(Reader reader, char separator, char quotechar, int line) {
095            this.br = new BufferedReader(reader);
096            this.separator = separator;
097            this.quotechar = quotechar;
098            this.skipLines = line;
099        }
100    
101        /**
102         * Reads the entire file into a List with each element being a String[] of
103         * tokens.
104         *
105         * @return a List of String[], with each String[] representing a line of the
106         *         file.
107         *
108         * @throws IOException
109         *             if bad things happen during the read
110         */
111        public List readAll() throws IOException {
112    
113            List allElements = new ArrayList();
114            while (hasNext) {
115                String[] nextLineAsTokens = readNext();
116                if (nextLineAsTokens != null)
117                    allElements.add(nextLineAsTokens);
118            }
119            return allElements;
120    
121        }
122    
123        /**
124         * Reads the next line from the buffer and converts to a string array.
125         *
126         * @return a string array with each comma-separated element as a separate
127         *         entry.
128         *
129         * @throws IOException
130         *             if bad things happen during the read
131         */
132        public String[] readNext() throws IOException {
133    
134            String nextLine = getNextLine();
135            return hasNext ? parseLine(nextLine) : null;
136        }
137    
138        /**
139         * Reads the next line from the file.
140         *
141         * @return the next line from the file without trailing newline
142         * @throws IOException
143         *             if bad things happen during the read
144         */
145        private String getNextLine() throws IOException {
146            if (!this.linesSkiped) {
147                for (int i = 0; i < skipLines; i++) {
148                    br.readLine();
149                }
150                this.linesSkiped = true;
151            }
152            String nextLine = br.readLine();
153            if (nextLine == null) {
154                hasNext = false;
155            }
156            return hasNext ? nextLine : null;
157        }
158    
159        /**
160         * Parses an incoming String and returns an array of elements.
161         *
162         * @param nextLine
163         *            the string to parse
164         * @return the comma-tokenized list of elements, or null if nextLine is null
165         * @throws IOException if bad things happen during the read
166         */
167        private String[] parseLine(String nextLine) throws IOException {
168    
169            if (nextLine == null) {
170                return null;
171            }
172    
173            List tokensOnThisLine = new ArrayList();
174            StringBuffer sb = new StringBuffer();
175            boolean inQuotes = false;
176            do {
177                    if (inQuotes) {
178                    // continuing a quoted section, reappend newline
179                    sb.append("\n");
180                    nextLine = getNextLine();
181                    if (nextLine == null)
182                        break;
183                }
184                for (int i = 0; i < nextLine.length(); i++) {
185    
186                    char c = nextLine.charAt(i);
187                    if (c == quotechar) {
188                            // this gets complex... the quote may end a quoted block, or escape another quote.
189                            // do a 1-char lookahead:
190                            if( inQuotes  // we are in quotes, therefore there can be escaped quotes in here.
191                                && nextLine.length() > (i+1)  // there is indeed another character to check.
192                                && nextLine.charAt(i+1) == quotechar ){ // ..and that char. is a quote also.
193                                    // we have two quote chars in a row == one quote char, so consume them both and
194                                    // put one on the token. we do *not* exit the quoted text.
195                                    sb.append(nextLine.charAt(i+1));
196                                    i++;
197                            }else{
198                                    inQuotes = !inQuotes;
199                                    // the tricky case of an embedded quote in the middle: a,bc"d"ef,g
200                                    if(i>2 //not on the begining of the line
201                                                    && nextLine.charAt(i-1) != this.separator //not at the begining of an escape sequence
202                                                    && nextLine.length()>(i+1) &&
203                                                    nextLine.charAt(i+1) != this.separator //not at the     end of an escape sequence
204                                    ){
205                                            sb.append(c);
206                                    }
207                            }
208                    } else if (c == separator && !inQuotes) {
209                        tokensOnThisLine.add(sb.toString());
210                        sb = new StringBuffer(); // start work on next token
211                    } else {
212                        sb.append(c);
213                    }
214                }
215            } while (inQuotes);
216            tokensOnThisLine.add(sb.toString());
217            return (String[]) tokensOnThisLine.toArray(new String[0]);
218    
219        }
220    
221        /**
222         * Closes the underlying reader.
223         *
224         * @throws IOException if the close fails
225         */
226        public void close() throws IOException{
227            br.close();
228        }
229    
230    }
