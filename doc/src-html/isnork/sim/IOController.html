027        public ArrayList<String> getPlayerList()
028        {
029            ArrayList<String> PlayerNames = new ArrayList<String>();
030    
031            try{
032                // Open the file that is the first
033                // command line parameter
034                FileInputStream fstream = new FileInputStream("Game.playerlist");
035                // Get the object of DataInputStream
036                DataInputStream in = new DataInputStream(fstream);
037                BufferedReader br = new BufferedReader(new InputStreamReader(in));
038                String strLine;
039                //Read File Line By Line
040                while ((strLine = br.readLine()) != null)   {
041                  // Print the content on the console
042                  //GameEngine.println (strLine);
043                  PlayerNames.add(strLine);
044                }
045                //Close the input stream
046                in.close();
047                }catch (Exception e){//Catch exception if any
048                  GameEngine.println("Error: " + e.getMessage());
049                }
050    
051    
052    
053            return PlayerNames;
054        }
055    
056        public Player getPObject(String playername)
057        {
058            Player myplayer = null;
059            try {
060                Class<Player> pc_object = (Class<Player>) Class.forName(playername);
061                 myplayer = (Player) pc_object.newInstance();
062    
063            } catch (InstantiationException ex) {
064                GameEngine.println("Player CLass " + playername + " could not be instantiated!");
065            } catch (IllegalAccessException ex) {
066                GameEngine.println("Player CLass " + playername + " Illegal Access!");
067            } catch (ClassNotFoundException ex) {
068                GameEngine.println("Player CLass " + playername + " was not found!");
069            }
070             return myplayer;
071        }
